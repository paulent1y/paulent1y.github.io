void setup() { // функция "настройки" запускается в начале программы один раз
  size(800, 600); // установка размеров экрана (ширина, высота) в пикселях
  //после того как задан экран можно задать переменные или что нибудь еще, что связано с размерами экрана
  a = width/2; //ширина пополам
}

void draw() { //функция запускающаяся бесконечно постоянно
  fill(20); //установка цвета заливки фигуры
  background(50); // установка цвета фона
  stroke(255); // установка цвета контура
  //цвета могут задаваться либо одним (числом), тогда цвет может быть выбран от 0 (черный) до 255(белый)
  //либо двумя (число, число) - тогда помимо цвета будет настроена прозрачность, тоже от 0(полностью прозрачный) до 255(непрозрачный)
  //либо тремя  (число, число, число) - тогда указывается каждый цветовой канал (red,green, blue)
  //либо четырьмя - (число, число, число, число) - четвертое число отвечает за прозрачность, первые три за rgb
  line(10,20,30,40); //рисует линию от точки (10,20) до (30, 40) за цвет линии отвечает stroke()
  rect(50,60, 70, 80); //рисует прямоугольник, который начинается в точке (50, 60), имеет ширину 70 и высоту 80
  ellipse(90, 100, 110, 120); //эллипс с центом в точке (90, 100) шириной 110 высотой 120
  circle(130, 140, 150); //круг в точке 130,140 с радиусом 150
  //все эти команды выполняются при каждом обновлении кадра 
}
