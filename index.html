import pygame, sys
from random import randint as r
pygame.init()

class Hero():
  def __init__(self):
    self.x = r(100,300)
    self.y = r(100,300)
    self.size = r(15,40)
    self.color = ( r(0,255),r(0,255),r(0,255) )
    self.vector = [ r(-5,5), r(-5,5)]
  def update(self):
    self.move()
    self.draw()
  def move(self):
    self.x += self.vector[0]
    self.y -= self.vector[1]

    if self.x width-self.size:
      self.vector[0] = - self.vector[0]
      self.x = width-self.size
    if self.y < self.size:
      self.vector[1] = -self.vector[1]
      self.y = self.size
    if self.y > height - self.size:
      self.vector[1] = -self.vector[1]
      self.y = height - self.size
  def draw(self):
    pygame.draw.circle(window, self.color, (int(self.x),int(self.y)), self.size) #отрисовка круга с заданными параметрами
